1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The user class
  8    *
  9    * @extends drupal.entity
 10    *
 11    * @param {object} object The user object.
 12    * @param {function} callback The function to be called once the user has
 13    * been retrieved from the server.
 14    */
 15   drupal.user = function(object, callback) {
 16 
 17     /** The name for this user. */
 18     this.name = '';
 19 
 20     /** The email address of our user. */
 21     this.mail = '';
 22 
 23     /** The password of the user. */
 24     this.pass = '';
 25 
 26     // Declare the api.
 27     this.api = new drupal.user.api();
 28 
 29     // Call the base class.
 30     drupal.entity.call(this, object, callback);
 31 
 32     // Make sure to also accept "uid" as the ID.
 33     this.id = object.uid || this.id;
 34   };
 35 
 36   // Derive from entity.
 37   drupal.user.prototype = new drupal.entity();
 38   drupal.user.prototype.constructor = drupal.user;
 39 
 40   /**
 41    * Returns the object to send to Services.
 42    */
 43   drupal.user.prototype.getObject = function() {
 44     return $.extend(drupal.entity.prototype.getObject.call(this), {
 45       name: this.name,
 46       mail: this.mail,
 47       pass: this.pass
 48     });
 49   };
 50 }(jQuery));
